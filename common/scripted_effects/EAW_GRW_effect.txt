#加载选择的部门档案
GRW_NAAP_set_active_department = {
	set_variable = {
		GRW_NAAP_active_department = temp_GRW_NAAP_department
	}

	clear_array = GRW_NAAP_current_entries
	meta_effect = {
		text = {
			for_each_loop = {
				array = GRW_NAAP_department_[ID_DEPARTMENT]_entries
				index = entry_index
				value = entry_value

				add_to_array = {
					array = GRW_NAAP_current_entries
					index = entry_index
					value = entry_value
				}
			}

			if = {
				limit = {
					check_variable = {
						temp_GRW_NAAP_entry > -1
					}
				}
				set_variable = {
					GRW_NAAP_active_entry = GRW_NAAP_department_[ID_DEPARTMENT]_entries^num
				}
				add_to_variable = {
					GRW_NAAP_active_entry = -1
				}
			}
			else = {
				set_variable = {
					GRW_NAAP_active_entry = -1
				}
			}
		}

		ID_DEPARTMENT = "[?temp_GRW_NAAP_department]"
	}

	if = {
		limit = {
			check_variable = {
				temp_GRW_NAAP_entry > -1
			}
		}

	}

	set_temp_variable = {
		temp_GRW_NAAP_department_to_activate = temp_GRW_NAAP_department
	}
	GRW_NAAP_set_active_department_frame = yes

	add_to_variable = {
		GRW_NAAP_dirty = 0.01
	}
}
