#大米略取（本国）
country_event = {
	id = GRW_flavor.2
	title = GRW_flavor.2.t
	desc = GRW_flavor.2.d
	picture = GFX_event_conquest_of_rice_book

	is_triggered_only = yes
	fire_only_once = yes

	option = {
		name = GRW_flavor.2.a
		hidden_effect = {
			every_other_country = {
				limit = {
					NOT = {
						has_government = communism
					}
					OR = {
						is_neighbor_of = GRW
						AND = {							
							is_major = yes
							communism > 0.06
							is_griffon_continent_griffon_country = yes
						}
					}
				}
				country_event = {
					id = GRW_flavor.5
					days = 30
				}
			}
		}
	}
}
